#bash

echo " This is a Python PIP package module installer for Windowzzzz"

echo " Last Updated April/3/2019"

echo "Upgrade PIP and install some basics."

python -m pip install --upgrade pip
python -m pip install setuptools
python -m pip install wheel
python -m pip install keyboard
python -m pip install serial
python -m pip install camera
python -m pip install joystick
python -m pip install pygame

echo "Install some math related things."

python -m pip install numpy
python -m pip install matplotlib
python -m pip install scipy

echo "Install Arduino and robot related stuff."

python -m pip install arduino
python -m pip install servo

echo "Install 3d Printer Drivers, CNC, STL, and Gcode related things."

python -m pip install pycnc
python -m pip install stl
python -m pip install qtgui
python -m pip install fabby
python -m pip install numpy-stl
python -m pip install pygrbl
python -m pip install grbl-link
python -m pip install grbl-gamepad

echo "Install Open Computer Vision and supporting things."

python -m pip install pyopengl
python -m pip install opencv-python
python -m pip install opencv-utils
python -m pip install YOSO

echo "Install Internet and Web related things."

python -m pip install beautifulsoup4

echo "Install Cryptography related stuff."

python -m pip install cryptography
